% --- Frame 6: Model + Punkty końcowe ---
\begin{frame}{Model statystyczny i punkty końcowe}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Bayesowski model Weibulla:}
            \[
                T_i \sim \mathrm{Weibull}(k,\, \lambda_i), \quad
                S(t \mid k, \lambda_i) = \exp\!\left[-\!\left(\tfrac{t}{\lambda_i}\right)^{\!k}\right]
            \]
            \[
                \log \lambda_i = \beta_0 + \alpha_{c(i)} + \gamma_{s(i)} + b_{g(i)}
            \]
            \vspace{0.2em}
            \textbf{Rozkłady a priori} (skalibrowane na pilocie):
            \[
                k \sim \mathrm{Gamma},\quad \beta_0,\,\alpha_c,\,\gamma_s \sim \mathcal{N},\quad \sigma_g \sim \mathrm{HN}
            \]
            Estymacja: HMC (\texttt{brms}/Stan), 4 łańcuchy $\times$ 1000 iter.
        \end{column}
        \begin{column}{0.47\textwidth}
            \textbf{Pierwszorzędowy:} ranking wg mediany a posteriori
            \[
                \tilde{m}_c = \mathrm{med}_s\bigl\{\exp(\beta_0^{(s)}\!+\alpha_c^{(s)})\cdot(\ln 2)^{1/k^{(s)}}\bigr\}
            \]
            \textbf{Drugorzędowe:} prawdopodobieństwo przeżycia
            \[
                \hat{S}_c(t) = \frac{1}{S}\sum_{s=1}^{S} \exp\!\left[-\!\left(\frac{t}{\lambda_c^{(s)}}\right)^{\!k^{(s)}}\right]
            \]
            \[
                t \in \{10,\,22\},\quad \lambda_c^{(s)} = \exp\!\bigl(\beta_0^{(s)}\!+\alpha_c^{(s)}\bigr)
            \]
            \footnotesize Wyniki raportowane z 95\% PrI.
        \end{column}
    \end{columns}
\end{frame}
